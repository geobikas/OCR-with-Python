# Get Kali Linux with WSL on Win11

    tesseract-ocr-ell tesseract-ocr-fra tesseract-ocr-tur tesseract-ocr-ara \
    imagemagick \
    libvips-tools \
    python3-tqdm \
    python3-pathlib
